│   │   ├── products/      # 产品管理
│   │   ├── blogs/         # 博客管理
│   │   ├── users/         # 用户管理

│   ├── layout.tsx         # 根布局
│   └── page.tsx           # 首页

│   ├── layout/           # 布局组件
│   │   ├── Header.tsx    # 网站头部
